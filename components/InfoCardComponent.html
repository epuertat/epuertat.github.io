import { Component, Input } from '@angular/core';

@Component({
  selector: 'cd-info-card',
  templateUrl: './info-card.component.html',
  styleUrls: ['./info-card.component.scss']
})
export class InfoCardComponent {
  @Input()
  cardTitle: string;
  @Input()
  link: string;
  @Input()
  cardClass = '';
  @Input()
  contentClass: string;
}

<div class="card"
     [ngClass]="cardClass">

      {{ cardTitle }}
    </ng-template>
  </div>
  <div class="card-body"

</div>
