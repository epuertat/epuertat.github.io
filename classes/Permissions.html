export class Permission {
  read: boolean;
  create: boolean;
  update: boolean;
  delete: boolean;

  constructor(serverPermission: Array<string> = []) {
    ['read', 'create', 'update', 'delete'].forEach(
      (permission) => (this[permission] = serverPermission.includes(permission))
    );
  }
}
